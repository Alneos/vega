176  bool validate(){
177  bool isValid = true;
178  std::vector<std::shared_ptr<T>> toBeRemoved;
179  for (iterator it = this->begin(); it != this->end(); ++it) {
180  std::shared_ptr<T> t = *it;
181  if (!t->validate()) {
182  isValid = false;
183  std::cerr << *t << " is not valid" << std::endl;
184 
185  switch (model.translationMode) {
186  case vega::ConfigurationParameters::MODE_STRICT:
187  // Shouldn't do any cleanup in STRICT mode
188  break;
189  case vega::ConfigurationParameters::MESH_AT_LEAST:
190  case vega::ConfigurationParameters::BEST_EFFORT:
191  toBeRemoved.push_back(t);
192  break;
193  default:
194  throw std::logic_error("Unknown enum in Translation mode");
195  }
196  }
197  }
198  for(auto& t: toBeRemoved) {
199  this->erase(*t);
200  }
201  return isValid;
202  };
