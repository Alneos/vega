93  /**
94  * Create a partition of materials and elements in the model, so that
95  * materials are assigned to elements.
96  * This is necessary for output languages such as Nastran.
97  * TODO: this is unimplemented at the moment.
98  */
99  const bool partitionModel;
100  const bool replaceDirectMatrices;
101  const bool removeRedundantSpcs;
102  /**
103  * Direct Matrices (DISCRETE_0D, DISCRETE_1D, STIFFNESS_MATRIX, MASS_MATRIX, DAMPING_MATRIX) may be
104  * too big to be handled by some solver (eg Systus).
105  * This bool commands the use of Model::splitDirectMatrices() in Model::finish(), which splits the matrices
106  * into smaller ones.
107  */
108  const bool splitDirectMatrices;
109  const int sizeDirectMatrices;
110  /**
111  * ElementSets of Direct Matrices (DISCRETE_0D, DISCRETE_1D, STIFFNESS_MATRIX, MASS_MATRIX, DAMPING_MATRIX)
112  * generally don't have associated cells/cellgroup. This may causes problems in generic element writer,
113  * such as in SYSTUS.
114  * This bool commands the use of Model::makeCellsFromDirectMatrices() in Model::finish(), which creates the
115  * needed cells.
116  */
117  const bool makeCellsFromDirectMatrices;
118  /**
119  * Build the corresponding cells to the LMPC constraints with a group, and a Rigid material.
120  * This bool commands the use of Model::makeCellsFromLMPC() in Model::finish(), which creates the
121  * needed cells.
122  */
123  const bool makeCellsFromLMPC;
124  /**
125  * Build the corresponding cells of RigidSets (RBAR, RBE3) with a group, and a Rigid material.
126  * This bool commands the use of Model::makeCellsFromRBE() in Model::finish(), which creates the
127  * needed cells.
128  */
129  const bool makeCellsFromRBE;
130  /**
131  * Some elementSet can hold very general elements, acting on various DOFs of the corresponding cells.
132  * For example, Nastran PELAS1 can regroup CELAS1 cells which "spring" in various DOFS 'DXtoDX, DYtoDZ, etc)
133  * whereas Systus can only have one spring direction by 1902 Part (the translation).
134  * This bool commands the use of Model::splitElementsByDOFS in Model::finish() which split the ElementSet
135  */
136  const bool splitElementsByDOFS;
137 };
