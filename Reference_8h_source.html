37  * An object that need to reference to a vega class can use this template to create a reference to an object of that class
38  */
39 template<class T>
40 class Reference final {
41 public:
42  template<class T2> friend std::string to_str(const Reference<T2>&);
43  template<class T2> friend std::ostream& operator<<(std::ostream&, const Reference<T2>&);
44  static constexpr int NO_ID = INT_MIN;
45  // TODO : make type, original_id and id private
46  typename T::Type type;
47  int original_id;
48  int id;
49  Reference(const typename T::Type type, const int original_id = NO_ID, const int id = NO_ID);
50  Reference(const T& t);
51  bool has_original_id() const {
52  return original_id != NO_ID;
53  }
54  bool has_id() const {
55  return id != NO_ID;
56  }
57  std::shared_ptr<Reference<T>> clone() const;
58  ~Reference() {
59  }
60  ;
61 };

73 template<class T>
74 bool operator==(const Reference<T>& left, const Reference<T>& right) {
75  if (left.type != right.type) {
76  return false;
77  }
78  if (left.original_id == T::NO_ORIGINAL_ID && right.original_id == T::NO_ORIGINAL_ID) {
79  return (left.id == right.id);
80  }
81  return left.original_id == right.original_id;
82 }
83 
84 template<class T>
85 bool operator <(const Reference<T>& left, const Reference<T>& right) {
86  if (left.type != right.type) {
87  return left.type < right.type; // first ordering : type
88  } else if (left.original_id != T::NO_ORIGINAL_ID && right.original_id != T::NO_ORIGINAL_ID) {
89  return left.original_id < right.original_id; // second ordering : original ids
90  } else if (left.original_id != T::NO_ORIGINAL_ID || right.original_id != T::NO_ORIGINAL_ID) {
91  return left.original_id != T::NO_ORIGINAL_ID; // third ordering : original_id wins over autoid
92  } else {
93  return left.id < right.id; // last ordering : compare autoids
94  }
95 }

97 template<class T>
98 std::string to_str(const Reference<T>& reference) {
99  std::ostringstream oss;
100  std::string id, type;
101 
102  if (reference.has_original_id())
103  id = "original_id=" + std::to_string(reference.original_id);
104  else
105  id = "vega_id=" + std::to_string(reference.id);
106 
107  auto it = T::stringByType.find(reference.type);
108  if (it != T::stringByType.end())
109  type = "type=" + it->second;
110  else
111  type = "unknown type";
112 
113  oss << "Reference[" << type << "; " << id << "]";
114 
115  return oss.str();
116 }

134 template<class T>
135 struct hash<vega::Reference<T>> {
136  size_t operator()(const vega::Reference<T>& reference) const {
137  size_t seed = 0;
138  // LD : hash should be different between, for example, Reference<LoadSet> and Reference<ConstraintSet>
139  // but LoadSet::Type and ConstraintSet::Type enums both begin with 0 value, so I use the string value instead.
140  boost::hash_combine(seed,hash<string>()(T::stringByType.find(reference.type)->second));
141  boost::hash_combine(seed,hash<int>()(reference.original_id));
142  boost::hash_combine(seed,hash<int>()(reference.id));
143  return seed;
144  }
