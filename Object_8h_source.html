37 template<class T> class Identifiable {
38  static int auto_id;
39  int original_id;
40  int id;
41 public:

43 
44  /**
45  * Were the Object in the original study ?
46  */
47  bool isOriginal() const {
48  return original_id != NO_ORIGINAL_ID;
49  }
50 
51  int getId() const {
52  return id;
53  }
54 
55  int bestId() const {
56  if (original_id != NO_ORIGINAL_ID) {
57  return original_id;
58  } else {
59  return id;
60  }
61  }

64  return original_id;
65  }

69  }

71  void resetId() {
72  id = ++auto_id;
73  original_id = NO_ORIGINAL_ID;
74  }
75 
76  Identifiable(int original_id = NO_ORIGINAL_ID) :
77  original_id(original_id), id(++auto_id) {
78  }

84  Reference<T> getReference() const {
85  int no_id = Reference<T>::NO_ID;
86  return Reference<T>(static_cast<const T * const >(this)->type,
87  (isOriginal() ? original_id : no_id), id);

90  virtual std::map<std::string, std::string> to_map() const {

109 template<class T>
110 std::string to_str(const T& t) {
111  std::ostringstream oss;
112  std::string type;
113 
114  auto it = T::stringByType.find(t.type);
115  if (it != T::stringByType.end())
116  type = it->second;
117  else
118  type = "unknown type";
119 
120  oss << T::name << "<" << type << ">" << "{";
121  bool first = true;
122  for (auto& kv : t.to_map()) {
123  if (!first)
124  oss << ";";
125  else
126  first = false;
127  oss << kv.first << "=" << kv.second;
128  }
129  oss << "}";
130 
131  return oss.str();
132 }
