175 private:

177 public:
178  DOFCoefs(double dx = 0, double dy = 0, double dz = 0, double rx = 0, double ry = 0,
179  double rz = 0) {
180  coefs[0] = dx;
181  coefs[1] = dy;
182  coefs[2] = dz;
183  coefs[3] = rx;
184  coefs[4] = ry;
185  coefs[5] = rz;
186  }
187  DOFCoefs& operator+=(const DOFCoefs& rv) {
188  for (int i = 0; i < 6; i++) {
189  coefs[i] += rv.coefs[i];
190  }
191  return *this;
192  }
193  double operator[](const int i) {
194  if (i < 0 || i > 5)
195  return 0; // TODO LD: why this case ???
196  return coefs[i];
197  }
198  bool operator<(const DOFCoefs& other) const;
199  bool operator==(const DOFCoefs& other) const;
